\section{RÈponse frÈquentielle}
\begin{frame}{DÈfinition}
	
	\begin{block}{RÈponse frÈquentielle}Ý
		Fonction qui permet d'analyser l'action du circuit sur un signal sinusoÔdal donnÈ selon sa frÈquence.
	\end{block}

	
	\begin{center}
		\includegraphics[width=0.7\linewidth]{figs/FR.pdf}
	\end{center}
	
	\[H(j\omega)=\frac{\bar{Y}(j\omega)}{\bar{X}}\]
	
\end{frame}

%--------------------------------------------------------------------------------
\begin{frame}{Exemple : impÈdance d'un circuit RL sÈrie}
	DÈfinissons l'entrÈe de notre systËme  
	$$x(t)=i(t)=I_m \cos(\omega t+ \phi_i),  \qquad \bar{X}=\bar{I}=I_m  e^{j\phi_i}$$
	et la sortie 
	$$ \bar{Y}(j\omega)=\bar{U}=Z(j\omega)\bar{I}=(R+j\omega L)\bar{I}.
	$$
	Dans ce cas la rÈponse frÈquentielle se rÈduit ý l'impÈdance du dipÙle : 
	\begin{gather*}
		H(j\omega)=\frac{\bar{Y}(j\omega)}{\bar{X}}=\frac{\bar{U}}{\bar{I}}=Z(j\omega)=R+j\omega L
	\end{gather*}
	Dans le domaine temporel : 
	\begin{align*}
		y(t)=u(t)& = \sqrt{R^2+\omega^2L^2}I_m \cos(\omega t +\phi_i+\arctan \frac{\omega L}{R})\\
		& = |H(j\omega)|I_m \cos(\omega t +\phi_i+\angle H(j\omega))
	\end{align*}
	\alert{$|H(j\omega)|$}: rÈponse en amplitude, modifie l'\textbf{amplitude} du signal $x(t)$.
	
	\alert{$\angle H(j\omega)$}: rÈponse en phase,  modifie la \textbf{phase} du signal $x(t)$.
\end{frame}
%--------------------------------------------------------------------------------

\begin{frame}{DÈfinition des 4 fonctions de transfert d'un quadripÙle}
	\begin{center}

		&H(j\omega)=\left.\frac{\bar{U}_2}{\bar{I}_1}\right|_{\bar{I}_2=0} \quad \text{impÈdance de transfert}\\
		&H(j\omega)=\left.\frac{\bar{I}_2}{\bar{U}_1}\right|_{\bar{U}_2=0} \quad \text{admittance de transfert}
	\end{align*}
	
\end{frame}

\begin{frame}{Forme analytique, pÙles et zÈros}
	
	Toute rÈponse frÈquentielle d'un circuit peut se mettre sous la forme rationnelle:
	\[H(j\omega)=\frac{N(j\omega)}{D(j\omega)}\]
	$N(j\omega)$ et $D(j\omega)$ sont deux polynomes en $j\omega$ ý coefficients rÈels.
	
	Les racines de $D(j\omega)=0$ sont les {\bf pÙles} de la fonction. Ils correspondent aux frÈquences naturelles du circuit.
	
	Les racines de $N(j\omega)=0$ sont les {\bf zÈros} de la fonction. A ces frÈquences :
	\[H(j\omega)=0 \quad \rightarrow \quad Y(j\omega)=0 \quad \rightarrow \quad \text{Elimination de cette frÈquence }\]
	
	Pour Èviter des calculs en nombres complexes, il est courant de remplacer la variable $j\omega$ par $s$ (formalisme de Laplace, cf. cours de signaux et systËmes).
	La fonction 
	\[H(s)=H(j\omega)|_{j\omega=s}\]
	est appelÈe {\bf fonction de transfert}.
\end{frame}

\begin{frame}{Le dÈcibel ou dB}
	
	\begin{block}{Le dÈcibel}
	…chelle logarithmique de mesure du rapport entre deux  puissances, ou gain en puissance 
	\[G_{dB}=10\log_{10}\frac{P_2}{P_1}\]
	\end{block}

	
	Si l'on exprime ce gain \textbf{en fonction de $U$ ou $I$}, 
	on obtient :
	\[P \propto U^2, I^2 \quad \Rightarrow G_{dB}=\mathbf{20}\log_{10}\frac{U_2}{U_1}= \mathbf{20}\log_{10}\frac{I_2}{I_1}\]
	
	Valeurs caractÈristiques :
	\begin{center}
		\begin{tabular}{ll|c}
			&& $G_{dB}$\\ \hline
			$P_1=P_2$ &  $U_1=U_2$ & 0\\
			$P_2=\frac{1}{2}P_1$ & $U_2=\frac{1}{\sqrt{2}}U_1$  & -3\\
			$P_2=\frac{1}{10}P_1$ & $U_2=\frac{1}{\sqrt{10}}U_1$  & -10\\ \hline
		\end{tabular} \hspace*{1cm}
		\begin{minipage}[c]{0.35\linewidth}
			Dans l'exemple prÈcÈdent, pour $\omega=\omega_0$
			\[\mathbf{20} \log_{10}\frac{|H(\omega_0)|}{|H(0)|}=-3 \, \text{dB}\]
		\end{minipage}
	\end{center}
	
	
\end{frame}

\subsection{ModËle idÈal}
\begin{frame}{ModËle idÈal}
	
	\begin{block}{Valeurs typiques des paramËtres d'un AO en pratique}
		\begin{itemize}
			\item $V_{sat} < 20$ V
			\item $A$ de l'ordre de $10^4$ ý $10^5$
			\item $\rightarrow$ dans la plage de fonctionnement linÈaire
			$|v_p-v_n|< 2$ mV
			\item $i_p$, $i_n$ de l'ordre de 0.1 mA
		\end{itemize}
	\end{block}
	
	\begin{block}{ModËle idÈal}
		\begin{itemize}
			\item $A=\infty \rightarrow \epsilon=0$ et $v_p=v_n$
			\item $i_p=i_n=0 \rightarrow Z_{in}=\infty$
		\end{itemize}
	\end{block}
	
\end{frame}

	\end{itemize}
\end{frame}

\begin{frame}{Contre rÈaction}
La \textit{contre-rÈaction}, plus communÈment appelÈe \textit{feedback}, provient du circuit qui ramËne ý l'entrÈe une fraction du signal de sortie : 

\begin{columns}
\begin{column}{0.5\textwidth}
\begin{center}
\includegraphics[width=0.6\textwidth]{figs/feedback}
\end{center}
\end{column}\hfill
\begin{column}{0.5\textwidth}
$E$ est le signal diffÈrentiel, $A$ le gain de la boucle directe, $H$ le gain de la boucle de retour ou de contre-rÈaction.
\end{column}
\end{columns}\vspace{2mm}

Le gain global "en boucle fermÈe" du systËme se dÈduit comme suit :
$E = V_i- HV_o$, $V_o = AE$, donc $V_o = AV_i -AHV_o$.
Ou encore 
\begin{equation}\label{eq:feedback_1}
\frac{V_o}{V_i} = \frac{A}{1+AH} = \frac{1}{1/A+H}.
\end{equation}
Pour un gain $A$ infini (cas de l'AO idÈal), l'expression (\ref{eq:feedback_1}) devient :
\begin{equation}
\frac{V_o}{V_i} = \frac{1}{H}.
\end{equation}
\textbf{Ceci montre comment, gr‚ce ý la contre-rÈaction, il est possible d'obtenir une tension de sortie finie avec un ÈlÈment de gain infini.}
\end{frame}

\begin{frame}{Dans la suite de cette leÁon ...}
Nous allons considÈrer quelques montages d'AO avec boucle de feedback et les caractÈriser.

\vfill

Nous faisons toujours l'hypothËse de l'AO idÈal.
\end{frame}

\section{Le montage amplificateur inverseur}
\subsection{CaractÈristiques}
\begin{frame}{Gain en tension}
	
	On cherche ý dÈterminer le gain en tension $\frac{v_o}{v_s}$ du montage. \vfill 
	
	\begin{columns}
		\begin{column}{0.5\linewidth}
			\begin{circuitikz} [scale=0.9]
				\draw (3.5,1.5) node[op amp] (opamp) {} % Opamp
				(opamp.+) node[left] {$v_p$}
				(opamp.-) node[below] {$v_n$};
				\draw (0,0) coordinate (A) % 
				to[american voltage source, l=$v_s$, i=$i_s$] (A |- opamp.-)
				to[R, l=$R_s$, -*] ++(1.5, 0) coordinate(VN);
				\draw (VN) to[short, i=$i_n$, -o] (opamp.-);
				\draw (VN) to[short, i=$i_f$] ++(0,1) coordinate (leftRf)
				to[R, l=$R_f$] (leftRf -| opamp.out) 
				to[short,-*] (opamp.out);
				\draw (opamp.out) to[short, o-o] ++(0.3,0) node[below] {$v_o$};
				\draw (0,0) to[short,-o] ++(5,0);
				\draw (opamp.+) to[short,o-*] (opamp.+ |- A) node[ground]{};
			\end{circuitikz}
		\end{column}
		\begin{column}{0.5\linewidth}
			PLK ý l'entrÈe inverseuse : \\ 
			\[i_s = i_f + i_n \text{ et } i_n=0\]
			\textbf{Masse virtuelle} ý l'entrÈe inverseuse :
			\[v_p=0 \, \Rightarrow \, v_n=0 \]
			De ces relations, on dÈduit que $$i_s=i_f=\frac{v_s}{R_s}$$
			et 
			$$v_o=-R_f i_f=-\frac{R_f}{R_s}v_s$$
		\end{column}
	\end{columns}
	Finalement, le gain vaut  
	\fbox{$\frac{\displaystyle v_o}{\displaystyle v_s}=-\frac{\displaystyle R_f}{\displaystyle R_s}$}

	\vfill
	
	Intuitivement : 
	"C'est la masse virtuelle ý l'entrÈe inverseuse, conjuguÈe ý la grande impÈdance d'entrÈe de l'AO, qui produisent l'effet d'amplification".
\end{frame}

\begin{frame}{Taux de contre-rÈaction}
	
	
	Le potentiel $v_o$ est fixÈ gr‚ce ý la boucle de retour (feedback)
	introduite par la rÈsistance $R_f$.
	
	\begin{block}{Taux de contre-rÈaction du montage}
		Fraction de la tension de sortie
		qui est ramenÈe aux bornes d'entrÈe de l'ampli-op. 
	\end{block}
	
	$i_n=0$ : tout ce passe comme si $R_s$ et $R_f$ Ètaient connectÈes en
	sÈrie. 
	
	\vfill 
	
	\begin{columns}
		\begin{column}{0.5\linewidth}

\begin{frame}{Choix de $R_s$ et $R_f$}
	
	Les rÈsultats prÈcÈdents sont valides seulement si l'AO se trouve dans
	sa plage de fonctionnement linÈaire. Pour cela, il faut :
	\[|v_o| < V_{sat},\quad \left|\frac{R_f}{R_s}v_s\right| < V_{sat},
	\quad \frac{R_f}{R_s} < \left|\frac{V_{sat}}{v_s}\right|\]
	
	Exemple :
	\begin{itemize}
		\item $V_{sat}=15$ V
		\item $v_s=10$ mV
		\item $\rightarrow  \frac{R_f}{R_s} < 1500$
	\end{itemize}
\end{frame}
